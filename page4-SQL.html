													TABLE schedule_date_clean
													AS
													SELECT * 
													FROM (
														SELECT 
															user_id,
															regexp_matches(dates, '\d{4}-\d{2}-\d{2}', 'g') AS extracted_dates,
															type,
															time_start,
															time_end,	
															timezone,	
															time_planned,	
															break_time,	
															leave_type	
														FROM schedule_copy
													) AS subq;

													SELECT 
													Unnest(extracted_id) AS user_id,
													Unnest(extracted_dates) AS dates,
													type,
													time_start,
													time_end,timezone,
													time_planned,
													break_time,
													leave_type
												FROM schedule_id_clean;
